import os
import re
from threading import Thread

import spaces
import gradio as gr
import torch
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    TextIteratorStreamer,
)

MODEL_ID = "lonewolf168/gemma-4-4b-khmer-sum"
MAX_INPUT_TOKENS = 4096

print("Loading tokenizer and model...")
tokenizer = AutoTokenizer.from_pretrained(MODEL_ID)

# ZeroGPU: load on CUDA up front. The `spaces` runtime makes the GPU
# available during import so weights live in GPU memory between calls.
model = AutoModelForCausalLM.from_pretrained(
    MODEL_ID,
    torch_dtype=torch.bfloat16,
    device_map="cuda" if torch.cuda.is_available() else "auto",
    attn_implementation="sdpa",
    low_cpu_mem_usage=True,
)
model.eval()

if tokenizer.pad_token_id is None:
    tokenizer.pad_token_id = tokenizer.eos_token_id

print("Model loaded.")


def get_prompt(text: str) -> str:
    return (
        "<start_of_turn>user\n"
        "You are an expert Khmer Synthesis Assistant. Your task is to analyze the provided text and "
        "provide the most appropriate professional summary in Khmer based on the content type.\n\n"
        "### INSTRUCTIONS:\n"
        "1. If the text is a MEETING TRANSCRIPT:\n"
        "   - Format the output as 'Professional Meeting Minutes'.\n"
        "   - Use these sections: 🎯 គោលបំណងនៃកិច្ចប្រជុំ (Objective), 💬 ចំណុចពិភាក្សាគន្លឹះ (Key Points), "
        "✅ សេចក្តីសម្រេច (Decisions), and 📝 កិច្ចការដែលត្រូវបន្ត (Action Items).\n\n"
        "2. If the text is a GENERAL DOCUMENT/ARTICLE:\n"
        "   - Format the output as a 'High-Accuracy Summary'.\n"
        "   - Provide a concise overview followed by a bulleted list of the most important factual points.\n\n"
        "### GUIDELINES:\n"
        "- Use formal, professional Khmer language.\n"
        "- Be factual, concise, and objective.\n"
        "- Do not add information that is not present in the text.\n\n"
        f"TEXT TO PROCESS:\n{text}\n"
        "<end_of_turn>\n"
        "<start_of_turn>model\n"
    )


_CLEAN_RE = re.compile(r"<start_of_turn>model|<start_of_turn>user|<start_of_turn>|<end_of_turn>")


def clean_gemma_output(text: str) -> str:
    if not text:
        return ""
    cleaned = _CLEAN_RE.sub("", text).strip()
    cleaned = re.sub(r"\n{3,}", "\n\n", cleaned)
    return cleaned


@spaces.GPU(duration=120)
def summarize(text: str, max_length: int):
    if not text or not text.strip():
        yield ""
        return

    prompt = get_prompt(text)
    inputs = tokenizer(
        prompt,
        return_tensors="pt",
        truncation=True,
        max_length=MAX_INPUT_TOKENS,
    ).to(model.device)

    streamer = TextIteratorStreamer(
        tokenizer,
        skip_prompt=True,
        skip_special_tokens=True,
        timeout=60.0,
    )

    generation_kwargs = dict(
        **inputs,
        streamer=streamer,
        max_new_tokens=int(max_length),
        do_sample=False,  # greedy: faster + deterministic for summarization
        use_cache=True,
        repetition_penalty=1.05,
        eos_token_id=tokenizer.eos_token_id,
        pad_token_id=tokenizer.pad_token_id,
    )

    thread = Thread(target=model.generate, kwargs=generation_kwargs)
    thread.start()

    accumulated = ""
    try:
        for new_text in streamer:
            accumulated += new_text
            yield clean_gemma_output(accumulated)
    finally:
        thread.join()


with gr.Blocks(title="Khmer Text Summarizer (Gemma 4B)") as iface:
    gr.Markdown(
        "# Khmer Text Summarizer (Gemma 4B)\n"
        "Paste a Khmer article or meeting transcript to generate a professional summary "
        "powered by `lonewolf168/gemma-4-4b-khmer-sum`. Running on ZeroGPU."
    )
    with gr.Row():
        with gr.Column():
            text_in = gr.Textbox(lines=14, label="Khmer Text to Summarize")
            max_len = gr.Slider(
                minimum=100,
                maximum=1024,
                value=512,
                step=10,
                label="Max Summary Length (Tokens)",
            )
            run_btn = gr.Button("Summarize", variant="primary")
        with gr.Column():
            text_out = gr.Textbox(lines=18, label="Summary Output")

    run_btn.click(fn=summarize, inputs=[text_in, max_len], outputs=text_out)
    text_in.submit(fn=summarize, inputs=[text_in, max_len], outputs=text_out)


if __name__ == "__main__":
    iface.queue(max_size=20).launch()
